rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // User profile photos
    match /profile_photos/{userId} {
      allow read: if true; // Public profiles
      allow write: if request.auth != null && request.auth.uid == userId;
    }
    
    // User uploaded images
    match /images/{userId}/{allPaths=**} {
      allow read: if true; // Images can be viewed by anyone with the link
      allow write: if request.auth != null && request.auth.uid == userId;
    }
    
    // PDF uploads - organized by block ID
    match /pdfs/{blockId}/{fileName} {
      allow read: if true; // PDFs can be viewed by anyone with the link
      allow write: if request.auth != null; // Any authenticated user can upload
      allow delete: if request.auth != null; // Any authenticated user can delete
    }
    
    // PDF thumbnails
    match /pdf-thumbnails/{blockId}/{fileName} {
      allow read: if true; // Thumbnails can be viewed by anyone
      allow write: if request.auth != null; // Any authenticated user can upload
      allow delete: if request.auth != null; // Any authenticated user can delete
    }
  }
}
